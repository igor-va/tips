Files in the root folder and subfolder
	- develop.dwg  # DWG development file
	- manual.dwg  # DWG assembly instructions file
	.pdf  # Various drawings for discussion
	- wiki.txt  # General technical information about the product or part being developed
	- tech.xlsx  # Specification, equipment, weight, length, strength
	- eng.docx  # Translations from Russian into English


Названия разрабатываемых деталей на сервере
	Detail Name - R2  # Профиль Крышка - R2
	Detail Name - R2 - Type  # Профиль Крышка - R2 - Сечение
	Project Name - Detail Name R2 - Type  # Профиль зажимной - Профиль Крышка R2 - Сечение


Названия разрабатываемых деталей во фьюжн
	Detail  # Крышка торцевая
	Subassembly - R2  # Подсборка - R2
	Assembly - R2  # Сборка R2
	Assembly - Subassembly - R2  # Сборка - Подсборка - R2
